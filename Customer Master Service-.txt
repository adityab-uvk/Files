              });
          }
          catch (Exception ex)
          {
              return StatusCode(500, new
              {
                  StatusCode = 500,
                  Message = "An error occurred while retrieving roles",
                  Error = ex.Message
              });
          }
      }

 // Get All Roles 
 public async Task<List<RoleDTO>> GetAllRolesAsync()
 {
     var roles = new List<RoleDTO>();

     try
     {
         using (SqlConnection conn = new SqlConnection(connectionString))
         {
             using (SqlCommand cmd = new SqlCommand())
             {
                 cmd.Connection = conn;
                 cmd.CommandType = CommandType.Text;

                 cmd.CommandText = "SELECT RoleId, RoleName FROM dbo.Roles ORDER BY RoleId";

                 await conn.OpenAsync();

                 using (SqlDataReader reader = await cmd.ExecuteReaderAsync())
                 {
                     while (await reader.ReadAsync())
                     {
                         roles.Add(new RoleDTO
                         {
                             RoleId = Convert.ToInt32(reader["RoleId"]),
                             RoleName = reader["RoleName"].ToString()
                         });
                     }
                 }
             }
         }

         return roles;
     }
     catch (Exception ex)
     {
         throw new ApplicationException("Error retrieving roles", ex);

     }
 }

// Get All Jobtitle

 public async Task<IEnumerable<JobTitleDTO>> GetAllJobTitleAsync()
 {
     var jobTitles = new List<JobTitleDTO>();

     try
     {
         using (SqlConnection conn = new SqlConnection(connectionString))
         {
             using (SqlCommand cmd = new SqlCommand("dbo.USP_GetAllJobTitles", conn))
             {
                 cmd.CommandType = CommandType.StoredProcedure;

                 await conn.OpenAsync();

                 using (SqlDataReader reader = await cmd.ExecuteReaderAsync())
                 {
                     while (await reader.ReadAsync())
                     {
                         jobTitles.Add(new JobTitleDTO
                         {
                             JobTitle = reader["JobTitle"].ToString()
                         });
                     }
                 }
             }
         }

         return jobTitles;
     }
     catch (Exception ex)
     {
         throw new ApplicationException("Error retrieving JobTitles", ex);
     }
 }

 public async Task<bool> UpdateRoleDetailsByJobTitleAsync(UpdateRoleDetailsDTO model)
 {
     try
     {
         using (SqlConnection conn = new SqlConnection(connectionString))
         {
             using (SqlCommand cmd = new SqlCommand("dbo.USP_UpdateRoleDetailsByJobTitle", conn))
             {
                 cmd.CommandType = CommandType.StoredProcedure;

                 cmd.Parameters.AddWithValue("@JobTitle", model.JobTitle);
                 cmd.Parameters.AddWithValue("@RoleId", model.RoleId);
                 cmd.Parameters.AddWithValue("@RoleDetails", model.RoleDetails ?? (object)DBNull.Value);

                 await conn.OpenAsync();

                 int rows = await cmd.ExecuteNonQueryAsync();

                 return rows > 0;
             }
         }
     }

     catch(Exception ex)
     {
         throw new ApplicationException("Error Updating Roles Details", ex);
     }
 }

ALTER PROCEDURE [dbo].[USP_GetAllJobTittles]
AS
BEGIN
    SET NOCOUNT ON;

    SELECT JobTitle
    FROM Users
    WHERE JobTitle IS NOT NULL 
          AND LTRIM(RTRIM(JobTitle)) <> ''
    GROUP BY JobTitle
    ORDER BY JobTitle
END

ALTER PROCEDURE [dbo].[USP_UpdateRoleDetailsByJobTitle]
    @JobTitle NVARCHAR(200),
    @RoleId INT,
    @RoleDetails NVARCHAR(500)
AS
BEGIN
    SET NOCOUNT OFF;

    UPDATE dbo.Users
    SET 
        RoleId = @RoleId,
        Role_Details = @RoleDetails,
        Updated_At = GETDATE()
    WHERE JobTitle = @JobTitle
END
